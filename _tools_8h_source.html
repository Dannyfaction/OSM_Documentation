100 T HermiteInterpolation(T from, T to, float t, float m0, float m1)
101 {
102  // Put t in [0,1]
103  if (t < 0)
104  t = 0;
105  else if (t > 1)
106  t = 1;
107 
108  // Interpolate
109  float t2 = t * t; // t squared
110  float t3 = t2 * t; // t cubbed
111  //
112  T val = (2 * t3 - 3 * t2 + 1) * from; // h00 * p0
113  val += (t3 - 2 * t2 + t) * m0; // h10 * m0
114  val += (-2 * t3 + 3 * t2) * to; // h01 * p1
115  val += (t3 - t2) * m1; // h11 * m1
116  //
117  return val;
118 }
